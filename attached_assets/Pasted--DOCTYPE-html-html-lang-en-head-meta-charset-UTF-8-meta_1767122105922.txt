    <script>
        const mockData = [
            { asset: 'EURUSD', strategy: 'SMC Breaker', session: 'London', condition: 'Trending', bias: 'Bullish', outcome: 'Win', rAchieved: 4.5, plAmt: 900, date: new Date().toISOString(), contextTF: 'D1', entryTF: 'M5' },
            { asset: 'NAS100', strategy: 'Silver Bullet', session: 'New York', condition: 'Trending', bias: 'Bearish', outcome: 'Win', rAchieved: 5.0, plAmt: 1500, date: new Date().toISOString(), contextTF: 'H4', entryTF: 'M1' },
            { asset: 'NAS100', strategy: 'Silver Bullet', session: 'New York', condition: 'Trending', bias: 'Bearish', outcome: 'Loss', rAchieved: -1, plAmt: -300, date: new Date().toISOString(), contextTF: 'H4', entryTF: 'M1' }
        ];

        let trades = JSON.parse(localStorage.getItem('matrix_journal_data_v7')) || mockData;

        function showTab(id) {
            ['dashboard', 'history', 'log'].forEach(t => document.getElementById(t).classList.add('hidden'));
            document.getElementById(id).classList.remove('hidden');
            updateUI();
        }

        function updateUI() {
            calculateGlobalStats();
            updateStratSelector();
            renderMatrices();
            renderStrategyDeepDrive();
            renderMarketIntelligence();
            renderHistory();
        }

        function updateStratSelector() {
            const selector = document.getElementById('granular-strat-selector');
            const currentVal = selector.value;
            const uniqueStrats = [...new Set(trades.map(t => t.strategy))];
            selector.innerHTML = `<option value="All">All Strategies</option>`;
            uniqueStrats.forEach(s => {
                const opt = document.createElement('option');
                opt.value = s; opt.innerText = s;
                selector.appendChild(opt);
            });
            selector.value = currentVal || "All";
        }

        function calculateGlobalStats() {
            const net = trades.reduce((a, b) => a + parseFloat(b.plAmt), 0);
            const wins = trades.filter(t => t.outcome === 'Win').length;
            const wr = trades.length ? Math.round((wins / trades.length) * 100) : 0;
            const exp = trades.length ? (trades.reduce((a, b) => a + parseFloat(b.rAchieved), 0) / trades.length).toFixed(2) : 0;
            const grossWin = trades.reduce((a, b) => a + (parseFloat(b.plAmt) > 0 ? parseFloat(b.plAmt) : 0), 0);
            const grossLoss = Math.abs(trades.reduce((a, b) => a + (parseFloat(b.plAmt) < 0 ? parseFloat(b.plAmt) : 0), 0));
            const pf = grossLoss > 0 ? (grossWin / grossLoss).toFixed(2) : grossWin.toFixed(2);

            document.getElementById('stat-profit').innerText = `$${net.toLocaleString()}`;
            document.getElementById('stat-wr').innerText = `${wr}%`;
            document.getElementById('stat-exp').innerText = `${exp}R`;
            document.getElementById('stat-count').innerText = trades.length;
            document.getElementById('stat-pf').innerText = pf;
        }

        function renderMarketIntelligence() {
            const selectedStrat = document.getElementById('granular-strat-selector').value;
            const filteredTrades = selectedStrat === "All" ? trades : trades.filter(t => t.strategy === selectedStrat);
            
            // Session Win Rate
            const sessWl = document.getElementById('session-wl-box');
            sessWl.innerHTML = '';
            ['London', 'New York', 'Asian'].forEach(s => {
                const sTrades = filteredTrades.filter(t => t.session === s);
                const wins = sTrades.filter(t => t.outcome === 'Win').length;
                const perc = sTrades.length ? Math.round((wins / sTrades.length) * 100) : 0;
                sessWl.innerHTML += `
                    <div>
                        <div class="flex justify-between text-[10px] font-black uppercase mb-1"><span>${s}</span><span class="text-blue-400">${perc}%</span></div>
                        <div class="w-full h-1.5 bg-slate-800 rounded-full overflow-hidden"><div class="h-full bg-blue-500" style="width: ${perc}%"></div></div>
                    </div>`;
            });

            // Bias Efficiency
            const biasWl = document.getElementById('bias-wl-box');
            biasWl.innerHTML = '';
            ['Bullish', 'Bearish'].forEach(b => {
                const bTrades = filteredTrades.filter(t => t.bias === b);
                const wins = bTrades.filter(t => t.outcome === 'Win').length;
                const perc = bTrades.length ? Math.round((wins / bTrades.length) * 100) : 0;
                biasWl.innerHTML += `
                    <div class="flex items-center justify-between p-3 bg-slate-900/40 rounded-xl border border-slate-800/50 mb-2">
                        <span class="text-[9px] font-black uppercase ${b === 'Bullish' ? 'text-blue-400' : 'text-rose-400'}">${b}</span>
                        <span class="text-xs font-black text-white">${perc}% Win</span>
                    </div>`;
            });

            // Streak Intervals
            const intBox = document.getElementById('interval-box');
            let maxWin = 0, maxLoss = 0, currentStreak = 0, streakType = 'None';
            if(filteredTrades.length > 0) {
                streakType = filteredTrades[filteredTrades.length-1].outcome;
                for(let i=filteredTrades.length-1; i>=0; i--) {
                    if(filteredTrades[i].outcome === streakType) currentStreak++; else break;
                }
                let w=0, l=0;
                filteredTrades.forEach(t => {
                    if(t.outcome === 'Win') { w++; l=0; if(w > maxWin) maxWin = w; }
                    else if(t.outcome === 'Loss') { l++; w=0; if(l > maxLoss) maxLoss = l; }
                });
            }
            intBox.innerHTML = `
                <div class="p-3 bg-slate-900/40 rounded-xl border border-slate-800/50 text-center">
                    <div class="text-[8px] text-slate-500 font-black uppercase">Current</div>
                    <div class="text-xl font-black ${streakType === 'Win' ? 'text-emerald-400' : 'text-rose-400'}">${currentStreak}</div>
                </div>
                <div class="p-3 bg-slate-900/40 rounded-xl border border-slate-800/50 text-center">
                    <div class="text-[8px] text-slate-500 font-black uppercase">Max Win</div>
                    <div class="text-xl font-black text-blue-400">${maxWin}</div>
                </div>`;
        }

        function renderStrategyDeepDrive() {
            const body = document.getElementById('stratDeepBody');
            body.innerHTML = '';
            const strats = [...new Set(trades.map(t => t.strategy))];
            strats.forEach(strat => {
                const sTrades = trades.filter(t => t.strategy === strat);
                const winTrending = sTrades.filter(t => t.condition === 'Trending' && t.outcome === 'Win').length;
                const totalTrending = sTrades.filter(t => t.condition === 'Trending').length;
                const winRanging = sTrades.filter(t => t.condition === 'Ranging' && t.outcome === 'Win').length;
                const totalRanging = sTrades.filter(t => t.condition === 'Ranging').length;
                
                body.innerHTML += `
                    <tr>
                        <td class="px-6 py-4 font-black text-slate-200">${strat}</td>
                        <td class="px-6 py-4 font-bold text-blue-400">${totalTrending ? Math.round((winTrending/totalTrending)*100)+'%' : '0%'}</td>
                        <td class="px-6 py-4 font-bold text-indigo-400">${totalRanging ? Math.round((winRanging/totalRanging)*100)+'%' : '0%'}</td>
                        <td class="px-6 py-4 text-center text-rose-500 font-black">---</td>
                        <td class="px-6 py-4 text-right font-black text-emerald-400">Stable</td>
                    </tr>`;
            });
        }

        function renderMatrices() {
            const computeRanking = (containerId, pk, sk) => {
                const container = document.getElementById(containerId);
                container.innerHTML = '';
                const primaries = [...new Set(trades.map(t => t[pk]))];
                primaries.forEach(pVal => {
                    const filtered = trades.filter(t => t[pk] === pVal);
                    const perf = {};
                    filtered.forEach(t => {
                        const sVal = t[sk];
                        if(!perf[sVal]) perf[sVal] = { profit: 0 };
                        perf[sVal].profit += parseFloat(t.plAmt);
                    });
                    const ranked = Object.entries(perf).sort((a,b) => b[1].profit - a[1].profit).slice(0, 3);
                    const wrap = document.createElement('div');
                    wrap.innerHTML = `<h5 class="text-[9px] font-black text-slate-500 uppercase mb-2 border-l-2 border-slate-700 pl-2">${pVal}</h5>`;
                    ranked.forEach(([name, stats], i) => {
                        wrap.innerHTML += `<div class="flex justify-between items-center mb-1 text-[11px]"><span class="truncate pr-2 text-slate-300 font-bold">#${i+1} ${name}</span><span class="${stats.profit >= 0 ? 'text-emerald-400' : 'text-rose-400'} font-black">+$${stats.profit.toFixed(0)}</span></div>`;
                    });
                    container.appendChild(wrap);
                });
            };
            computeRanking('rank-strat-inst', 'asset', 'strategy');
            computeRanking('rank-strat-sess', 'session', 'strategy');
            computeRanking('rank-sess-inst', 'asset', 'session');
            computeRanking('rank-tf-inst', 'asset', 'entryTF');
        }

        function renderHistory() {
            const body = document.getElementById('historyBody');
            body.innerHTML = '';
            [...trades].reverse().forEach(t => {
                body.innerHTML += `<tr class="border-b border-slate-800/30"><td class="px-6 py-4 font-black text-white">${t.asset}</td><td class="px-6 py-4 text-slate-400 font-bold uppercase text-[10px]">${t.strategy}</td><td class="px-6 py-4 text-center"><span class="px-2 py-1 rounded text-[9px] font-black ${t.outcome === 'Win' ? 'bg-emerald-500/10 text-emerald-400' : 'bg-rose-500/10 text-rose-400'}">${t.outcome}</span></td><td class="px-6 py-4 text-right font-black ${t.plAmt >= 0 ? 'text-emerald-400' : 'text-rose-400'}">$${parseFloat(t.plAmt).toFixed(0)}</td></tr>`;
            });
        }

        document.getElementById('tradeForm').addEventListener('submit', (e) => {
            e.preventDefault();
            const data = { date: new Date().toISOString(), contextTF: 'D1', entryTF: 'M5' };
            new FormData(e.target).forEach((val, key) => data[key] = val);
            // Manual gathering since FormData needs 'name' attributes
            const manual = ['asset', 'strategy', 'session', 'condition', 'bias', 'outcome', 'rAchieved', 'plAmt'];
            manual.forEach(id => data[id] = document.getElementById(id).value);
            trades.push(data);
            localStorage.setItem('matrix_journal_data_v7', JSON.stringify(trades));
            e.target.reset();
            showTab('dashboard');
        });

        window.onload = () => showTab('dashboard');
